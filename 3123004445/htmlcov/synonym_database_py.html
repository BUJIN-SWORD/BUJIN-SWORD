62def get_synonyms(word: str) -> List[str]: 

63 """ 

64 获取词语的所有同义词（整合四大词库，优先匹配语境） 

65 :param word: 待查询词语 

66 :return: 包含自身的同义词列表 

67 """ 

68 # 整合所有词库 

69 all_synonyms = {**THEME_SYNONYMS, **ROLE_SYNONYMS, **SCENE_SYNONYMS, **OBJECT_SYNONYMS} 

70 

71 # 精确匹配词库 

72 for base_word, synonyms in all_synonyms.items(): 

73 if word == base_word or word in synonyms: 

74 return synonyms.copy() 

75 

76 # 若未匹配到，返回自身（确保兼容性） 

77 return [word]
